function _wap_tracking_get_vars() {
  $temp = $_SERVER['QUERY_STRING'];

  if(!empty($parsed)) {
    $queryParameters = $temp;
    if(strpos($queryParameters, 'df_name', 0) != false){
      $queryParameters .= "&donation_currency=GBP"; 
    }
  } else {
    $queryParameters = "''";
  }
  return $queryParameters;
}
